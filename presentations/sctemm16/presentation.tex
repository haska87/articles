\section{Математическая модель}
\subsection{Многогрупповое диффузионное приближение}
\begin{frame}{Многогрупповое диффузионное приближение}
Динамика нейтронного потока рассматривается в ограниченной выпуклой 2D или 3D области  $\Omega$ с границей $\partial \Omega$. Перенос нейтронов описывается системой уравнений:
\[
\begin{split}
 \frac{1}{v_g} \frac{\partial \phi_g}{\partial t} - & \nabla \cdot D_g \nabla \phi_g + \Sigma_{rg} \phi_g 
 - \sum_{g\neq g'=1}^{G} \Sigma_{s,g'\rightarrow g} \phi_{g'} = \\
 =  & \ (1-\beta) \chi_g \sum_{g'=1}^{G} \nu \Sigma_{fg'} \phi_{g'} + \widetilde{\chi}_g \sum_{m=1}^{M} \lambda_m c_m , \quad 
 g = 1,2, ..., G .
\end{split}
\]
Здесь $\phi_g$ -- поток нейтронов,
$v_g$ -- эффективная скорость,\\
$D_g$ -- коэффициент диффузии, 
$\Sigma_{rg}$ -- сечение увода,\\
$\Sigma_{s,g'\rightarrow g}$ -- сечение рассеяния,
$\nu\Sigma_{fg}$ -- сечение генерации,\\
$\beta$ -- эффективная доля запаздывающих нейтронов (ЗН),\\ 
$\chi_g$, $\widetilde{\chi}_g$  -- спектры МН и ЗН, 
$c_m$ -- плотность источников ЗН,\\  
$\lambda_m$ -- постоянная распада источников ЗН.
\end{frame}

\begin{frame}{Запаздывающие нейтроны}
Плотность источников запаздывающих нейтронов описывается уравнениями:
\[
 \frac{\partial c_m}{\partial t} + \lambda_m c_m = \beta_m \sum_{g=1}^{G} \nu \Sigma_{fg} \phi_g,
 \quad m = 1,2, ..., M, 
\]
где $\beta_m$ --- доля запаздывающих нейтронов  $m$ типа, причем:
\[
 \beta = \sum_{m=1}^{M} \beta_m .
\] 

\end{frame}

\begin{frame}{Без учета запаздывающих нейтронов}
Без учета запаздывающих нейтронов можно ограничиться задачей для нейтронного потока:
\[
\begin{split}
 \frac{1}{v_g} \frac{\partial \phi_g}{\partial t} - & \nabla \cdot D_g \nabla \phi_g + \Sigma_{rg} \phi_g 
 - \sum_{g\neq g'=1}^{G} \Sigma_{s,g'\rightarrow g} \phi_{g'} = \\
 =  & ((1-\beta)\chi_g + \beta\widetilde{\chi}_g) \sum_{g'=1}^{G} \nu \Sigma_{fg'} \phi_{g'} , \quad 
 g = 1,2, ..., G .
\end{split}
\]
\end{frame}

\begin{frame}{Начальные и граничные условия}
На границе области $\partial \Omega$ ставятся условия альбедного типа:
\[
 D_g\frac{\partial \phi_g}{\partial n} + \gamma_g \phi_g = 0, \quad 
 \quad g = 1,2, ..., G ,
\]
где $n$ -- внешняя нормаль к границе $\partial \Omega$, $\gamma_g$ -- групповой альбедный параметр.
\vfill

Рассматривается краевая задача с граничными условиями альбедного типа, и начальными условиями:
\[
\phi_g(\bm x,0) = \phi_g^0(\bm x), 
\quad  g = 1,2, ..., G ,
\quad   c_m(\bm x,0) = c_m^0(\bm x), 
\quad  m = 1,2, ..., M .
\]
\end{frame}

\subsection{Операторная формулировка}
\begin{frame}{Операторная формулировка}
Обозначения
\[
\begin{split}
\bm \phi = &\ \{\phi_1, \phi_2, ..., \phi_G\},
\bm c = \{c_1, c_2, ..., c_M\} \\
V = &\ (v_{g g'}), \quad v_{g g'} = \delta_{g g'} v_g^{-1}, 
\\
D = &\ (d_{g g'}), \quad d_{g g'} = - \delta_{g g'} \nabla \cdot D_g \nabla, 
\\
S = &\ (s_{g g'}), \quad  s_{g g'} = \delta_{g g'} \Sigma_{rg} - \Sigma_{s,g'\rightarrow g}, 
\\
R = &\ (r_{g g'}), \quad  r_{g g'} = (1-\beta)\chi_g \nu \Sigma_{fg'}, 
\\
\widetilde{R} = &\ (\widetilde{r}_{gg'}), \quad \widetilde{r}_{gg'} = ((1-\beta)\chi_g + \beta\widetilde{\chi}_g)\nu\Sigma_{fg'},
\\
B = &\ (b_{g m}), \quad b_{g m} = \widetilde{\chi}_g \lambda_m, 
\\
\Lambda = &\ (\lambda_{m m'}), \quad  \lambda_{m m'} = \lambda_m \delta_{m m'}, 
\\
Q = &\ (q_{mg}), \quad  q_{mg} = \beta_m \nu \Sigma_{fg},
\\
g, g' = &\ 1,2, ..., G, \quad m, m' = 1,2, ....,M,  
\end{split}
\]
где
\[
 \delta_{g g'} = \left \{ 
 \begin{matrix}
 1, & g = g', \\
 0, & g \neq  g'.
 \end{matrix}
 \right . 
\]
\end{frame}

\begin{frame}{Операторная формулировка}
С учетом введенных обозначений система уравнений с учетом запаздывающих нейтронов записывается в следующем виде:
\[
\begin{split}
V \frac{d \bm \phi}{d t} + (D+S) \bm \phi &= R \bm \phi + B\bm c,
\\
\frac{d \bm c}{d t} + \Lambda \bm c &= Q \bm \phi. 
\end{split}
\]
Без учета запаздывающих нейтронов:
\[
 V \frac{d \bm \phi}{d t} + (D+S) \bm \phi = \widetilde{R} \bm \phi.
\]
Рассматривается задача Коши при следующих начальных и граничных условиях:
\begin{eqnarray*}
\bm \phi(0) = \bm \phi^0, \quad  \bm c(0) = \bm c^0,
\qquad
D\frac{d \bm\phi}{d n} + L \bm \phi = 0,
\end{eqnarray*}  
где $L = (l_{g g'}), \quad l_{g g'} = \delta_{g g'} \gamma_g$.
\end{frame}


\subsection{Спектральные задачи}
\begin{frame}{Спектральные задачи}
С учетом запаздывающих нейтронов она формулируется в виде:
\[
\begin{split}
(D+S) \bm \varphi  &= \mu  (R \bm \varphi + B \bm s),\\
\Lambda \bm s &= \mu Q \bm \varphi .
\end{split}
\]
Аналогично, без учета запаздывающих нейтронов:
\[
\begin{split}
(D+S) \bm \varphi  &= \mu \widetilde{R} \bm \varphi.\\
\end{split}
\]
Минимальное собственное значение:
\[
 k = \frac{1}{\mu^{(1)}}  
\] 
есть эффективный коэффициент размножения.
Значение $k = \mu^{(1)} = 1$ связывается с критическим состоянием реактора, соответствующая
собственная функция $\bm \varphi^{(1)}(\bm x)$ есть стационарное решение уравнения.
\end{frame}

\begin{frame}{Регулярный режим}
Более приемлемая спектральная характеристика связана с $\alpha$-спектральной задачей.
С учетом запаздывающих нейтронов она имеет вид
\[
\begin{split}
 (D+S - R) \bm \varphi  -  B \bm s & = \mu V \bm \varphi , \\
 \Lambda \bm s - Q \bm \varphi & =  \mu   \bm s .
\end{split} 
\]
Без учета запаздывающих нейтронов имеем
\[
(D + S - \widetilde{R})\bm \varphi  =  \mu V \bm \varphi.
\]
Главное собственное значение
\[ 
 \alpha = \mu^{(1)}
\]
называется $\alpha$--собственным значением (период реактора). \\
С собственным значением $\alpha$ 
можно связать асимптотическое поведение
решения задачи Коши  при больших временах. 
В этом регулярном режиме поведение реактора описывается функцией $e^{-\alpha t} \bm\varphi^{(1)}(\bm x)$, 
то есть поведение решения определяется только первой гармоникой. 
\end{frame}

\section{Дискретизация}
\subsection{Аппроксимация по времени}
\begin{frame}{Аппроксимация по времени}
Определим сетку
$\omega = \{ t^n=n \tau, \quad n = 0,1,...,N, \quad \tau N = T \}$ и используем обозначения $\bm\phi^n = \bm\phi(\bm{x}, t^n)$, $\bm c^n = \bm c(\bm{x}, t^n)$.\\

Используется численно-аналитический метод:
\[
  \frac{\partial e^{\lambda_m t}  c_m}{\partial t} = \beta_m  e^{\lambda_m t} \sum_{g=1}^{G} \nu \Sigma_{fg} \phi_g,
 \quad m = 1,2, ..., M .
\] 
Интегрирование от $t^{n}$ до $t^{n+1}$ дает:
\begin{eqnarray*}
c_m^{n+1} = e^{-\lambda_m\tau} c_m^n + \beta_m \int_{t_n}^{t_{n+1}}e^{\lambda_m (t-t^{n+1})} \sum_{g=1}^{G} \nu \Sigma_{fg} \phi_g d t,
 \quad m = 1,2, ..., M.
\end{eqnarray*}
\end{frame}

\begin{frame}{Без учета запаздывающих нейтронов}
Чисто неявная схема:
\[
V\frac{\bm\phi^{n+1}-\bm\phi^n}{\tau}+(D+S)\bm\phi^{n+1} = \widetilde{R} \bm\phi^{n+1}.
\]
Явно-неявная схема:
\[
V\frac{\bm\phi^{n+1}-\bm\phi^n}{\tau}+(D+S)\bm\phi^{n+1} = \widetilde{R} \bm\phi^{n}.
\]
Схема Кранка-Николсон:
\[
V\frac{\bm\phi^{n+1}-\bm\phi^n}{\tau}+(D+S)\frac{\bm\phi^{n+1} + \bm\phi^{n}}{2} = \widetilde{R} \frac{\bm\phi^{n+1} + \bm\phi^{n}}{2}.
\]
\end{frame}

\begin{frame}{С учетом запаздывающих нейтронов}
Чисто неявная схема:
\[
\begin{split}
V \frac{\bm\phi^{n+1}-\bm\phi^n}{\tau}+(D+S) 
\bm \phi^{n+1} &= R \bm \phi^{n+1} + B\bm c^{n+1},
\\
\bm{c}^{n+1} & = \widetilde{\Lambda}\bm{c}^{n} + \tau Q \bm{\phi}^{n+1},
\end{split}
\]
где
\[
\widetilde{\Lambda} = (\widetilde{\lambda}_{mm'}), \quad \widetilde{\lambda}_{mm'} = \delta_{mm'} e^{-\lambda_m\tau},
 \quad m, m' = 1,2, ....,M .
\]
Явно-неявная схема:
\[
\begin{split}
V \frac{\bm{\phi}^{n+1} - \bm{\phi}^{n}}{\tau}+(D+S)\bm{\phi}^{n+1} &= R\bm{\phi}^{n} + B\bm{c}^{n+1} , \\
\bm{c}^{n+1} & = \widetilde{\Lambda}\bm{c}^{n} + \tau \widetilde{Q} \bm{\phi}^{n},
\end{split}
\]
где 
\[
\widetilde{Q} = (\widetilde{q}_{mg}), \quad \widetilde{q}_{mg}=e^{-\lambda_m\tau}\beta_m\nu\Sigma_{fg}, 
 \quad g = 1,2, ..., G,
 \quad m = 1,2, ....,M .
\]
\end{frame}

\end{split}
\]
при всех $\bm \xi  \in V^D, \ \bm \zeta  \in V^M$.

\end{frame}

\subsection{Выход на регулярный режим}
\begin{frame}{Выход на регулярный режим}
Начальные условия для всех схем:
\[ 
\phi_1^0 = 1.0, \quad \phi_2^0 = 0.25
\]
при  $k=24$, $p=2$. Пусть $T=5\times 10^{-3}$ и рассмотрим решение   чисто неявной схемы при $\tau = 1 \times 10^{-5}$ как квазиточное решение. \\
Выход на регулярный режим будем контролировать с помощью оценки близости отнормированного решения нестационарной задачи и главной собственной функции $\bm \phi$. Для первой группы положим:
\[
 \eta(t) = \|\overline{\phi}_1(t) - \overline{\varphi}_1\|, 
 \quad  \overline{\phi}_1(t) = \frac{\phi_1(t)}{\|\phi_1(t)\|} ,
 \quad  \overline{\varphi}_1 = \frac{\varphi_1}{\|\varphi_1\|} .
\] 
Установление темпа динамического развития оценивается величиной:
\[
\theta(t) = \frac{1}{\sqrt{2} \norm{\alpha}} \left ( 
\norm{\frac{1}{\phi_1(t)}\frac{\partial\phi_1(t)}{\partial t} - \alpha}^2 +
\norm{\frac{1}{\phi_2(t)}\frac{\partial\phi_2(t)}{\partial t} - \alpha}^2 \right )^{1/2}.
\]
\end{frame}

\section{Заключение}
\begin{frame}{Заключение}
\begin{footnotesize}
\begin{itemize}
\item
Рассмотрена нестационарная задача диффузии нейтронов в ядерном реакторе при использовании многогруппового приближения. 
Выделена $\alpha$-спектральная задача, которая характеризует
динамическое нейтронное поле ядерного реактора  на асимптотической стадии при больших временах -- регулярный режим.
\item
Контроль точности приближенного решения проводиться на последовательности сгущающихся сеток с использованием конечных элементов различной степени.
\item
Рассмотрена нестационарная задача диффузии нейтронов с выходом на регулярный режим в двух случаях: с учетом ЗН и без учета ЗН. 
\item
Тестовые расчеты выполнены в двумерном приближении для ядерного реактора ВВВЭР-1000 без отражателя с использованием двухгруппового диффузионного приближения. Установлена хорошая отделимость собственных значений в $\alpha$-спектральной задаче с учетом и без учета ЗН. 
\item
Наблюдается хорошая сходимость нестационарной задачи при уменьшении шага по времени для чисто неявной схемы. Явно-неявная схема сходится намного хуже,  чем чисто неявная схема. Схема Кранка-Николсон хоть и имеет второй порядок аппроксимации, практически непригодна для моделирования регулярного режима ядерного реактора. 

\end{itemize}
\end{footnotesize}
